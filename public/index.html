import time
import requests

api_url = f"{window.location.origin}/bakadevApi/v1/chack-md5"
payload = {
    "md5": "665c3365a0278eb22e6a8bce4c1652", 
    "expectedAmount": 7.20,"expectedCurrency": "USD"              
}

while True:
    try:
        response = requests.post(api_url, json=payload).json()
        status = response.get("status")
        
        if status == "VERIFIED_SUCCESS":
            print("🎉 ទទួលបានលុយគ្រប់ចំនួនហើយ!")
            break
        elif status == "AMOUNT_MISMATCH":
            print("⚠️ អតិថិជនវេរលុយមកខុសចំនួន!")
            break
        else:
            print(f"⏳ {response.get('message')}...")
            time.sleep(4)
    except Exception as e:
        time.sleep(4)
