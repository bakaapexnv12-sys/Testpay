import time
import requests

# ១. កំណត់ព័ត៌មានទឹកប្រាក់ និង MD5 ដែលបានមកពី KHQR
api_url = f"{window.location.origin}/bakadevApi/v1/chack-md5"
payload = {
    "md5": "d60f3db96913029a2af979a1662c1e72",  # លេខ MD5 របស់ QR
    "expectedAmount": 0.01,                 # ចំនួនលុយត្រូវបង់ពិតប្រាកដ
    "expectedCurrency": "USD",              # ប្រភេទលុយ USD ឬ KHR
    "webhookUrl": "https://bot.com/webhook"  # (លុយចូលហើយ ឱ្យបាញ់ទៅប្រាប់ Bot)
}

# ២. បង្កើត Loop រត់ឆែកក្នុង Bot (ឆែករៀងរាល់ ៤ វិនាទីម្តង ការពារកុំឱ្យគាំង)
print("កំពុងរង់ចាំអតិថិជនស្កេនទូទាត់...")
while True:
    try:
        response = requests.post(api_url, json=payload).json()
        
        if response.get("status") == "VERIFIED_SUCCESS":
            print("🎉 ទទួលបានលុយត្រឹមត្រូវហើយ! ប្រព័ន្ធនឹងបង្កើតសមតុល្យ (Balance) ជូនភ្លាមៗ។")
            break
        elif response.get("status") == "AMOUNT_MISMATCH":
            print("⚠️ ព្រមាន៖ អតិថិជនលួចកែបន្លំចំនួនលុយបង់!")
            break
        else:
            time.sleep(4)
    except Exception as e:
        print(f"មានបញ្ហាភ្ជាប់ទៅកាន់ API: {e}")
        time.sleep(4)
